RHI Texture 使用文档

1. 模块概述

RHI Texture 子系统提供了跨图形 API 的 2D 纹理抽象层，支持：

  • 多种纹理格式：UNorm、Float 等多种格式
  • 自动 Mipmap 生成：支持运行时生成 mipmap 链
  • HDR 支持：支持浮点纹理数据
  • 灵活的采样器状态：Wrap 模式和 Filter 模式可配置
  • 多种纹理类型：1D、2D、3D、CubeMap

2. 核心类型

2.1 TextureDimension 枚举

枚举值 说明
Tex1D 一维纹理
Tex2D 二维纹理
Tex3D 三维纹理
Cube 立方体贴图

2.2 Format 枚举（常用）

枚举值 说明
R8_UNorm 单通道 8 位无符号归一化
RG8_UNorm 双通道 8 位无符号归一化
RGB8_UNorm 三通道 8 位无符号归一化
RGBA8_UNorm 四通道 8 位无符号归一化
RGBA8_SRGB 四通道 sRGB 色彩空间
R16_Float 单通道 16 位浮点
RGBA16_Float 四通道 16 位浮点（HDR）
RGBA32_Float 四通道 32 位浮点

2.3 TextureUsage 枚举

枚举值 说明
None 无用途
ShaderResource 着色器资源（采样）
RenderTarget 渲染目标
DepthStencil 深度模板
Storage 存储纹理（Shader 读写）
TransferSrc 拷贝源
TransferDst 拷贝目标

2.4 WrapMode 枚举

枚举值 说明
Repeat 重复模式
ClampToEdge 夹紧到边缘
MirroredRepeat 镜像重复
ClampToBorder 夹紧到边框

2.5 FilterMode 枚举

枚举值 说明
Nearest 最近邻采样
Linear 线性过滤
NearestMipmapNearest 最近邻 + 最近 mipmap
LinearMipmapNearest 线性 + 最近 mipmap
NearestMipmapLinear 最近邻 + 线性 mipmap
LinearMipmapLinear 三线性过滤

2.6 TextureDesc 结构

struct TextureDesc {
    TextureDimension dimension = TextureDimension::Tex2D;
    Format format = Format::RGBA8_UNorm;

    uint32_t width = 1;
    uint32_t height = 1;
    uint32_t depth = 1;

    uint32_t mipLevels = 1;
    uint32_t arrayLayers = 1;
    uint32_t sampleCount = 1;

    TextureUsage usage = TextureUsage::ShaderResource;

    bool generateMips = false;
    bool flipVerticalOnLoad = true;

    WrapMode wrapS = WrapMode::Repeat;
    WrapMode wrapT = WrapMode::Repeat;
    WrapMode wrapR = WrapMode::Repeat;

    FilterMode minFilter = FilterMode::Linear;
    FilterMode magFilter = FilterMode::Linear;
};

3. 基本用法

3.1 创建 RHI 设备

#include "Render/Public/RHI/RHI_device.h"

auto device = Render::RHI::CreateOpenGLDevice();

3.2 从文件加载 2D 纹理（推荐）

#include "Render/Public/RHI/RHI_device.h"
#include "Render/Public/RHI/RHI_texture.h"

using namespace Render::RHI;

// 使用 RHI 设备加载纹理
auto texture = device->LoadTexture2D("path/to/texture.png", true);
if (!texture) {
    std::cerr << "Failed to load texture!" << std::endl;
    return;
}

// 获取纹理参数
const auto& desc = texture->GetDesc();
std::cout << "Width: " << desc.width << std::endl;
std::cout << "Height: " << desc.height << std::endl;
std::cout << "Mip Levels: " << desc.mipLevels << std::endl;
std::cout << "Format: " << static_cast(desc.format) << std::endl;

3.3 手动创建纹理

// 创建纹理描述
TextureDesc desc;
desc.dimension = TextureDimension::Tex2D;
desc.format = Format::RGBA8_UNorm;
desc.width = 1024;
desc.height = 1024;
desc.mipLevels = 1;
desc.usage = TextureUsage::ShaderResource;

// 创建纹理资产
auto textureAsset = device->CreateTexture(desc);
if (!textureAsset) {
    std::cerr << "Failed to create texture!" << std::endl;
    return;
}

// 准备纹理数据
std::vector pixelData(1024 * 1024 * 4, 255); // 白色纹理

// 上传数据
auto* glTexture = dynamic_cast(textureAsset.get());
if (glTexture && glTexture->Check()) {
    glTexture->SetData(pixelData.data());
}

3.4 绑定纹理到着色器

// 绑定纹理到指定单元
auto* glTexture = dynamic_cast(textureAsset.get());
if (glTexture) {
    glTexture->Bind(0); // 绑定到纹理单元 0
}

// 在着色器中使用
// uniform sampler2D u_Texture;
// glActiveTexture(GL_TEXTURE0);
// glBindTexture(GL_TEXTURE_2D, textureHandle);

3.5 生成 Mipmaps

auto* glTexture = dynamic_cast(textureAsset.get());
if (glTexture) {
    glTexture->GenerateMips();
}

4. 完整测试程序示例

#include <iostream>
#include <glad/glad.h>
#include <GLFW/glfw3.h>
#include "Render/Public/RHI/RHI_device.h"
#include "Render/Public/RHI/RHI_texture.h"
#include "Render/Private/Backend/Opengl/gl_texture.h"

using namespace Render::RHI;

int main() {
    // 初始化 GLFW
    if (!glfwInit()) {
        return -1;
    }

    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 5);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

    GLFWwindow* window = glfwCreateWindow(800, 600, "RHI Texture Test", nullptr, nullptr);
    if (!window) {
        glfwTerminate();
        return -1;
    }

    glfwMakeContextCurrent(window);

    if (!gladLoadGLLoader(reinterpret_cast<GLADloadproc>(glfwGetProcAddress))) {
        std::cerr << "Failed to initialize GLAD" << std::endl;
        return -1;
    }

    // 创建 RHI 设备
    auto device = CreateOpenGLDevice();
    if (!device) {
        std::cerr << "Failed to create OpenGL device" << std::endl;
        return -1;
    }

    // 加载纹理
    std::string texturePath = "path/to/texture.png";
    auto texture = device->LoadTexture2D(texturePath, true);
    if (!texture) {
        std::cerr << "Failed to load texture from: " << texturePath << std::endl;
        return -1;
    }

    // 输出纹理参数
    const auto& desc = texture->GetDesc();
    std::cout << "\n=== Texture Parameters ===" << std::endl;
    std::cout << "Width: " << desc.width << " pixels" << std::endl;
    std::cout << "Height: " << desc.height << " pixels" << std::endl;
    std::cout << "Depth: " << desc.depth << std::endl;
    std::cout << "Mip Levels: " << desc.mipLevels << std::endl;
    std::cout << "Dimension: 2D" << std::endl;
    std::cout << "Format: RGBA8_UNorm" << std::endl;
    std::cout << "Usage: ShaderResource" << std::endl;
    std::cout << "Generate Mips: " << (desc.generateMips ? "yes" : "no") << std::endl;

    // 验证 OpenGL 纹理
    auto* glTexture = dynamic_cast<Render::Backend::OpenGL::GLTexture*>(texture.get());
    if (glTexture) {
        GLuint handle = glTexture->GetHandle();
        std::cout << "\nOpenGL Verification:" << std::endl;
        std::cout << "  Texture Handle: " << handle << std::endl;
        std::cout << "  glIsTexture: " << (glIsTexture(handle) ? "true" : "false") << std::endl;
    }

    std::cout << "\n[SUCCESS] RHI Texture Loading Test PASSED!" << std::endl;

    // 清理
    glfwDestroyWindow(window);
    glfwTerminate();

    return 0;
}

7. 注意事项

重要提示：
  1. OpenGL 上下文：加载纹理前必须确保 OpenGL 上下文已激活
  2. 文件路径：纹理文件路径相对于可执行文件所在目录
  3. 线程安全：当前实现不是线程安全的，需要在渲染线程中操作
  4. 资源生命周期：确保在设备销毁前销毁所有纹理
  5. Mipmap 生成：生成 mipmap 会增加显存使用，但能改善远距离渲染质量
  6. 垂直翻转：默认情况下图像会在加载时垂直翻转（符合 OpenGL 惯例）

8. 支持的图像格式

基于 stb_image 库，支持以下常见格式：

  • PNG - 便携式网络图形（支持透明度）
  • JPEG - 有损压缩照片格式
  • BMP - 位图格式
  • PSD - Photoshop 文档
  • GIF - 图形交换格式
  • HDR - 高动态范围图像
  • TGA - Truevision TGA

9. 扩展说明

当前实现支持 OpenGL 4.5+ 后端，未来可扩展：

  • D3D11/D3D12 后端：实现相同的 RHI 接口
  • Vulkan 后端：使用 VkImage 和 VkImageView
  • Metal 后端：使用 MTLTexture
  • 纹理数组：支持 TextureArray
  • 压缩纹理：支持 BC/DXT/ETC 压缩格式
  • 纹理流送：支持大规模纹理的渐进式加载
